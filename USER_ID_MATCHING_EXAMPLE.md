def get_user_id_by_email(self, email: str) -> str:
    """Fetch 6-digit User ID from Users sheet"""
    
    all_rows = users_worksheet.get_all_values()
    
    for row in all_rows[1:]:  # Skip header
        if row[1] == email:  # Column B = Email
